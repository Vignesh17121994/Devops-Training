1. run-instances :

Example 1: To launch an instance :

# Multi-Line Commannd :

aws ec2 run-instances \
    --image-id "ami-08df646e18b182346" \
    --instance-type t2.micro \
    --count 1 \
    --subnet-id "subnet-5183ac39" \
    --key-name "awskeys-mum" \
    --security-group-ids "sg-0aa1782637f5f7dd9" \
    --user-data file://hostname.txt

# One Line Command :
aws ec2 run-instances --image-id "ami-08df646e18b182346" --instance-type t2.micro --count 1 --subnet-id "subnet-5183ac39" --key-name "awskeys-mum" --security-group-ids "sg-0aa1782637f5f7dd9" --user-data file://hostname.txt
